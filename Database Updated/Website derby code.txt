CREATE TABLE Category(
Category_Name VARCHAR(255) NOT NULL,
PRIMARY KEY (Category_Name)
);

CREATE TABLE Threads(
Thread_title VARCHAR(255) NOT NULL,
Thread_message VARCHAR(5000) NOT NULL,
Thread_Date DATE NOT NULL,
Category_Name VARCHAR(255) NOT NULL,
PRIMARY KEY (Thread_title),
FOREIGN KEY (Category_Name) REFERENCES Category (Category_Name)
);

CREATE TABLE Posts(
Post_ID INT NOT NULL,
Post_title VARCHAR(255) NOT NULL,
Post_date DATE NOT NULL,
Posts_message VARCHAR (5000) NOT NULL,
Post_URL VARCHAR(2083),
Category_Name VARCHAR(255) NOT NULL,
PRIMARY KEY (Post_ID),
FOREIGN KEY (Category_Name) REFERENCES Category (Category_Name)
);

CREATE TABLE postComment(
Comment_message VARCHAR (5000) NOT NULL,
Comment_Date DATE NOT NULL,
Thread_title VARCHAR(255) NOT NULL,
Post_ID INT NOT NULL,
PRIMARY KEY (Comment_message, Comment_Date),
FOREIGN KEY (Thread_title) REFERENCES Threads (Thread_title),
FOREIGN KEY (Post_ID) REFERENCES Posts (Post_ID)
);
